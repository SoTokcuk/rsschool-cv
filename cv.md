## Code Example 
```C++
long reduction(long numerator, long denominator)
{
    long gcd = abs(denominator);
    if (numerator < 0)
    {
        numerator *= -1;
    }
    if (numerator > denominator)
    {
        while ((numerator % gcd != 0) || (denominator % gcd != 0))
        {
            gcd--;
        }
    }
    else
    {
        gcd = 1;
    }
    return gcd;
}
```
